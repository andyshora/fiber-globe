# Add Countries to Mesh

1. Import and add `<ThreeCountries />` to mesh.

```diff
// three-mesh.js

import React from 'react';

import ThreeGraticule from './three-graticule'
+ import ThreeCountries from './three-countries'


const ThreeMesh = () => {
  return (
    <mesh>
      <sphereGeometry args={[1, 32]} />
      <meshPhongMaterial color="#191919" transparent={true} opacity={0.8} />
      <ThreeGraticule />
+     <ThreeCountries />
    </mesh>
  );
};

export default ThreeMesh;
```
